
#[derive(PartialEq, Eq, Debug)]
struct Problem {
    seeds: Vec<u32>,
    seed_to_soil: Vec<Vec<u32>>,
}

fn parse_input(s: &str) -> IResult<&str, Input> {
    let (input, _) = tag("seeds: ")(s)?;
    let (input, seeds) = list(input)?;

    let (input, _) = tag("\n\n")(input)?;
    let (input, _) = tag("seed-to-soil map:\n")(input)?;
    let (input, groups) = matrix(input)?;

    Ok((
        input,
        Problem {
            seeds,
            seed_to_soil: groups,
        },
    ))
}
